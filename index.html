High Order Function

Uma função recebe outra função como parâmetro

function soma(x, y) {
  return x + y;
}
function multiplicacao(x, y) {
  return x + y;
}
function calcula(op, x, y) {
  return op(x, y);
}
calcula(soma, 2, 3); // 5
calcula(multiplicacao, 2, 3); // 6

High Order Function

Uma função devolve outra função como resultado

function adder(x) {
  return function(y) {
    return x + y;
  }
}

var add5 = adder(5);

add5(10); // 15

Closure

Escopo e contexto

function saudacao(mensagem) {
